## Step 2：实现评测运行器

```python
# eval/runner.py
import json
import time
from dataclasses import dataclass
from typing import List

@dataclass
class EvalCase:
    id: str
    category: str
    difficulty: str
    user_input: str
    success_criteria: List[str]
    forbidden: List[str]
    expected_tools: List[str] = None

class EvalRunner:
    def __init__(self, agent_factory, judge, parallel: int = 4):
        self.agent_factory = agent_factory
        self.judge = judge
        self.parallel = parallel
    
    def load_suite(self, path: str) -> List[EvalCase]:
        with open(path) as f:
            data = json.load(f)
        return [EvalCase(**c) for c in data["cases"]]
    
    def run(self, suite_path: str) -> dict:
        cases = self.load_suite(suite_path)
        print(f"📋 Loaded {len(cases)} test cases")
        
        # 并行评测
        evaluator = ParallelEvaluator(self.agent_factory, self.parallel)
        results = evaluator.evaluate(cases)
        
        # 分类汇总
        summary = self._summarize(results)
        
        # 打印报告
        print(generate_eval_report(summary))
        
        return summary
    
    def _summarize(self, results: list) -> dict:
        total = len(results)
        success = sum(1 for r in results if r["task_success"])
        by_cat = {}
        by_diff = {}
        
        for r in results:
            cat = r["category"]
            diff = r["difficulty"]
            
            for d, store in [(cat, by_cat), (diff, by_diff)]:
                if d not in store:
                    store[d] = {"total": 0, "success": 0}
                store[d]["total"] += 1
                if r["task_success"]:
                    store[d]["success"] += 1
        
        return {
            "suite_name": "TravelAgent Eval Suite v1",
            "total_cases": total,
            "success_rate": success / total,
            "by_category": {
                c: {"success_rate": d["success"] / d["total"], "total": d["total"]}
                for c, d in by_cat.items()
            },
            "by_difficulty": {
                d: {"success_rate": v["success"] / v["total"], "total": v["total"]}
                for d, v in by_diff.items()
            },
            "failures": [r for r in results if not r["task_success"]],
            "avg_latency_ms": sum(r["latency_ms"] for r in results) / total,
        }


# 使用
if __name__ == "__main__":
```

## Step 3：CI 配置

```yaml
# .github/workflows/agent-eval.yml
name: Agent Eval
on:
  pull_request:
    paths:
      - 'src/agent/**'
      - 'src/prompts/**'
      - 'src/tools/**'

jobs:
  eval:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: "3.12" }
      - run: pip install -r requirements.txt
      
      - name: Run Eval Suite
        run: python -m eval.runner --suite tests/eval_suite.json
        env:
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
      
      - name: Post Report to PR
        if: github.event_name == 'pull_request'
        run: |
          cat reports/latest.md >> $GITHUB_STEP_SUMMARY
```

## Step 4：持续改进循环

```
评测不是一次性的，是一个循环：

  ① 跑评测 → 发现成功率 82%（不达标）
  ② 分析 failures → 发现 hard 类"模糊需求"全崩
  ③ 改进 Agent → 增加追问逻辑
  ④ 跑评测 → 成功率 88%（达标）✅
  ⑤ 添加新的 edge cases → 防止类似问题回归
  ⑥ 回到 ①

每一次循环，评测用例集 +1，Agent 能力 +1。
```

---

## Step 5：最终检查清单

```
□ 评测用例集 ≥ 50 条
□ 覆盖 3+ 类别（规划/查询/预订/异常）
□ 覆盖 easy/medium/hard 三个难度
□ 每条有 success_criteria
□ 配置了 GitHub Actions
□ 门禁规则清晰（min success_rate = 0.85）
□ 每次 PR 自动跑评测并评论报告
□ eval_history 表在增长
```
